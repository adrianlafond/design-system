public static delta(r1: number, r2: number): number {
  const a1 = Angle.normalize(r1);
  const a2 = Angle.normalize(r2);
  let delta = a2 - a1;
  if (Math.abs(delta) > Math.PI) {
    delta = a2 - Math.PI * 2 - a1;
  }
  return delta;
}
